# MAC git提交流程
# 新建分支
git branch newbranch
# 查看当前分支
git branch
# 切换到newbranch分支
git checkout newbranch
# 在newbranch中添加修改文件
git add .
git commit -m "push info"
# 切换到master分支
git checkout master
# 合并newbranch到master
git merge newbranch
# 提交
git push - u origin master
# 删除分支
git branch -D newbranch
